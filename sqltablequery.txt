create table accounts(
    id int AUTO_INCREMENT PRIMARY KEY,
    user_name  varchar(30) unique,
    mobile_no varchar(10),
    pass varchar(30),
    team varchar(30),
    email_id varchar(30)
    );


    create table deposite(
    id int PRIMARY key AUTO_INCREMENT,
    mobile_no varchar(10),
    user_name varchar(30) unique,
    deposite bigint,
    team varchar(30) ,
    last_paid_date Date  
    );

create table penalty(
        id int PRIMARY key AUTO_INCREMENT,
        user_name varchar(30),
        penalty int,
        team varchar(30),
        Date varchar(10),
        note text
    );
create table expendature(
        id int PRIMARY key AUTO_INCREMENT, 
        expendature int,
        team varchar(30),
        Date varchar(10),
        note text
 );

 create table loan(
     id int PRIMARY key AUTO_INCREMENT,
     user_name varchar(30),
     team varchar(10),
     loan_amt bigint,
     loan_amt_returned bigint,
     loan_amt_intrest int,
     loan_amt_intrest_returned int,
     loan_type varchar(15),
     EMI_amt int,
     EMI_duration int,
     EMI_count int,
     EMI_rate float(2),
     Loan_date varchar(10),
     status varchar(15),
     last_paid_date varchar(10),
     loan_provider varchar(30)
 );

 Create table cashbook(
    id int PRIMARY key AUTO_INCREMENT,
    user_name varchar(30),
    amount bigint,
    note text,
    team varchar(30)
);

Create table adminname(
    id int PRIMARY key AUTO_INCREMENT,
    team varchar(30),
    admin_name varchar(30),
    admin_mobile_no varchar(10),
    admin_password varchar(50),
    permision int
    
);

create table remuneration(
        id int PRIMARY key AUTO_INCREMENT, 
        remuneration int,
        team varchar(30),
        Date varchar(10),
        note text
 );
